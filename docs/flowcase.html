CVPartner scraper

This is a package for scraping the cvpartner api

Getting started

Install this fork of cvpartnerpy from this repo: pip install git+https://github.com/eiriks/cvpartnerpy.git

Usage

The package has support for getting user metadata and CVs attached to users. This can be used to scrape out all user CVs from the api:

import os
from flowcase import Flowcase

cvp = Flowcase(org="myorg", api_key=os.environ["FLOWCASE_API_KEY"])

Search for single CV

from IPython.display import HTML, display
from cvpartner.types.search_result import SearchItem

# Find a CV to work on:
results = cvp.search_users(query="Rufus Scrimgeour")

print(f"{len(results)} users found")
table=[["name", "user_id", "cv_id"]]
for user in results:
    user: SearchItem
    #print(user.name, "\t\t\t\t|", user.user_id, "\t|", user.default_cv_id)
    table.append([user.name, user.user_id, user.default_cv_id])

display(HTML(
    "<table><tr>{}</tr></table>".format(
        "</tr><tr>".join(
            "<td>{}</td>".format("</td><td>".join(str(_) for _ in row)) for row in table)
    )
))

Retrieve a spesific CV.

user_cv = cvp.get_user_cv(user_id="5a16db...b862",
                          cv_id="5a1...eb863")

Retrieve a full department:

DEPARTMENT_NAME = "Department of Magical Accidents and Catastrophes"

department = cvp.get_emploees_and_cvs_from_department(DEPARTMENT_NAME)

208    def initiate_report(self) -> str:
209        request_data = {
210            "offset": 0,
211            "size": 100,
212            "must": [],
213            "workflow_stage_filter": "live",
214            "filter_owned": False,
215            "output_format": "xlsx",
216        }
217        # print(self._auth_header)
218        # print(request_data)
219        url = REFERENCE_REPORTS_API_V4_URL.format(org=self.org)
220        response = requests.post(url, data=request_data, headers=self._auth_header)
221        if response.status_code == 200:
222            return response.json()["id"]  # ["_id"]
223        else:
224            raise Exception(f"Failed to initiate report: {response.status_code}")
225
226    # Trinn 2: Pulle for å sjekke om rapporten er ferdig
227    def pull_report_status(self, report_id: str) -> str:
228        """Pull to check if the report is finished."""
229        url = f"{REFERENCE_REPORTS_API_V4_URL.format(org=self.org)}/{report_id}"
230        log.debug(f"Polling report status at {url}")
231
232        while True:
233            response = requests.get(url, headers=self._auth_header)
234            report_data = response.json()
235
236            percent_processed = report_data.get("percentProcessed", 0)
237            state = report_data.get("status", "started")
238
239            log.debug(f"Report processing: {percent_processed}%")
240
241            if state == "finished":
242                file_url = report_data.get("file").get("url")
243                if file_url:
244                    log.debug("Report is finished!")
245                    return file_url
246                else:
247                    raise Exception("Report finished, but no file URL found.")
248            log.debug(report_data)
249            log.debug("Waiting 5s for report to finish...")
250            time.sleep(5)  # Wait 5 seconds before next polling
251
252    # Trinn 3: Last ned rapportfilen fra den signerte URL-en
253    def download_report_file(self, file_url, output_path):
254        response = requests.get(file_url)
255
256        if response.status_code == 200:
257            with open(output_path, "wb") as f:
258                f.write(response.content)
259            print(f"Report downloaded successfully: {output_path}")
260        else:
261            raise Exception(
262                f"Error downloading file: {response.status_code}, {response.text}"
263            )
264
265    # fiannlay do all the things
266    # get todays year as YYYY and month as MM and day as DD in a string
267
268    def get_reference_report(self, output_filename: Optional[str] = None) -> str:
269        report_id = self.initiate_report()
270        file_url = self.pull_report_status(report_id)
271
272        # make a filename with todays date
273        today = datetime.datetime.now().strftime("%Y-%m-%d")
274        if not output_filename:
275            output_filename = f"reference_report_{today}.xlsx"
276        else:
277            output_filename = f"{output_filename}_{today}.xlsx"
278
279        self.download_report_file(file_url, output_filename)
280        return output_filename
